use Bookie_DB;


create table Genres (
BookId int unique not null auto_increment primary key,
Fantasy boolean,
Adventure boolean,
Dystopian boolean,
PoliticalFiction boolean,
Fiction boolean,
ComingOfAge boolean,
SciFi boolean
);

create table Books (
ID int unique not null primary key auto_increment,
Author varchar(255),
Title varchar(255),
Description varchar(255),
Price decimal(10,2),
QuantityAvailable int,
ImageUrl blob,
foreign key (ID) references Genres(BookId)
);
